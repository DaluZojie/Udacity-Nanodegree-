/* Query 1- query used for first Insight */SELECT m.MediaTypeId,m.Name as MediaType, count( m.Name)  as Total_countFROM Track tJOIN MediaType mON m.MediaTypeId = t.MediaTypeIdGroup by m.NameOrder by m.MediaTypeId /* Query 2- query used for the second Insight */SELECT a.Name, count(al.ArtistId) as Number_of_Album, al.TitleFROM Album alJOIN Artist aON al.ArtistId = a.ArtistIdGroup by a.Name Order by Number_of_Album DESCLimit 12/* Query 3- query used for the third Insight */SELECT c.LastName as Customer, Sum(i.Total) as TotalFROM Invoice iJOIN Customer cON i.CustomerId=c.CustomerId Group by  Customer Order by Total DESClimit 10/* Query 4- query used for the fourth Insight */SELECT e.LastName || " "|| e.FirstName as Support_Rep,  count(c.FirstName || " "|| c.LastName) as Number_of_CustomersFROM Employee eJOIN Customer cON e.EmployeeId = c.SupportRepIdGroup by Support_Rep
